function	script	Func_Casing	{
	mes "[凯尼]";
	mes "现在, 你可以用";
	mes "500 个"+getitemname(getarg(0));
	mes "以及 500 Zeny 换一个 "+getitemname(getarg(1))+", 所以请确定";
	mes "您带够了子弹";
	mes "和 Zeny 来兑换. ";
	next;
	mes "[凯尼]";
	mes "你可以用 500 个 "+getitemname(getarg(0))+"和 500 Zeny 来兑换";
	mes "一个 "+getitemname(getarg(1))+". ";
	next;
	mes "[凯尼]";
	mes "请记得我一次最多只能";
	mes "帮您兑换 50 个弹夹. ";
	mes "请输入您想要兑换";
	mes "的弹夹数量. ";
	mes "如果想取消兑换, 请输入 ''0. ''";
	next;
	input .@amount;
	if (.@amount > 50 || .@amount == 0) {
		mes "[凯尼]";
		mes "嘿, 我不能给您";
		mes "那么多弹夹. ";
		mes "请重试, 不要";
		mes "输入一个大于";
		mes "50 的数字. ";
		close;
	}
	if (countitem(getarg(0)) >= (500*.@amount)) {
		if (Zeny >= (500*.@amount)) {
			if (checkweight(getarg(1),.@amount) == 0) {
				mes "[凯尼]";
				mes "嘿, 您的背包里";
				mes "塞的东西太多了. ";
				mes "清理出一些空间, ";
				mes "然后再来用子弹";
				mes "和我换弹夹, ";
				mes "明白了吗? ";
				close;
			}else{
				mes "[凯尼]";
				mes "很好, 这里是您的";
				mes "弹夹. 谢谢光临我的";
				mes "商店, 同时我希望您";
				mes "能好好利用手里的这";
				mes "些弹药. ";
				set Zeny, Zeny - 500*.@amount;
				delitem getarg(0),500*.@amount;
				getitem getarg(1),.@amount;
				close;
			}
		} else {
			mes "[凯尼]";
			mes "对不起, 您没有足够的";
			mes "zeny 来换取这些弹夹. ";
			mes "等你存够了钱, ";
			mes "再回到我的商店, ";
			mes "那时候, ";
			mes "我再帮你换弹夹. ";
			close;
		}
	} else {
		mes "[凯尼]";
		mes "对不起, 但是您没有";
		mes "足够的弹药来兑换";
		mes "这个弹夹. ";
		mes "也许您需要再检查";
		mes "一下您的物品栏. ";
		mes "确定了再来找我. ";
		close;
	}
}
