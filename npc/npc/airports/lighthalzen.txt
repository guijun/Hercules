lhz_airport,143,43,5	script	机场 工作人员#lhz_air1a::lhz_airport1	4_F_01,{
	mes "[机场 工作人员]";
	mes "欢迎来到";
	mes "里希塔乐镇机场,";
	mes "这里我们提供前往";
	mes "艾音布洛克、毁葛以及朱诺的直达航班.";
	next;
	if (select("登机:取消") == 1) {
		mes "[机场 工作人员]";
		mes "登机费用是";
		mes "1,200 Zeny, 但是您可以";
		mes "凭一张飞空艇免费使用券";
		mes "免去费用.";
		next;
		if (select("是:否") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "lhz_airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "lhz_airport",148,51;
				end;
			}
			mes "[机场 工作人员]";
			mes "抱歉, 您的金钱不足";
			mes "1,200 zeny 以支付";
			mes "登机费用.";
			close;
		}
	}
	mes "[机场 工作人员]";
	mes "谢谢, ";
	mes "欢迎下次再来.";
	mes "祝您愉快~";
	close;
}
