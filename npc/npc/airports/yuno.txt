y_airport,143,43,5	script	机场 工作人员#y_air1a::y_airport1	4_F_01,{
	mes "[机场 工作人员]";
	mes "欢迎来到朱诺机场, 在这里我们提供飞往艾音布洛克, 里希塔乐镇以及毁葛的国内航班, ";
	mes "以及前往伊斯鲁德与拉赫的国际航班.";
	mes "有什么我能帮助您的吗?";
	next;
	if (select("登机:取消") == 1) {
		mes "[机场 工作人员]";
		mes "乘坐飞空艇的费用";
		mes "是 1,200 zeny.如果您";
		mes "使用一张飞空艇免费使用券, ";
		mes "那就可以免登机费.您要乘坐吗?";
		next;
		if (select("是:否") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "y_airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "y_airport",148,51;
				end;
			}
			mes "[机场 工作人员]";
			mes "很抱歉, ";
			mes "您的钱不足 1,200 zeny";
			mes "来付登机费.";
			close;
		}
	}
	mes "[机场 工作人员]";
	mes "谢谢!";
	mes "祝您愉快.";
	close;
}
