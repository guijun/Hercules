airport,143,43,5	script	机场 工作人员#airport1a::airport1	4_F_01,{
	mes "[机场 工作人员]";
	mes "欢迎来到";
	mes "艾音布洛克机场, ";
	mes "这里我们提供前往";
	mes "朱诺, 里希塔乐镇以及毁葛的";
	mes "直达航班.";
	next;
	if (select("乘坐飞艇:取消") == 1) {
		mes "[机场 工作人员]";
		mes "坐飞艇的费用是";
		mes "1,200 zeny, 但是如果您有";
		mes "飞空艇乘坐免费券, ";
		mes "这笔费用可以不收.";
		mes "您要乘坐飞空艇吗?";
		next;
		if (select("是:否") == 1) {
			if (countitem(7311) > 0) {
				delitem 7311,1; //Free_Flying_Ship_Ticket
				warp "airport",148,51;
				end;
			}
			if (Zeny >= 1200) {
				set Zeny, Zeny - 1200;
				warp "airport",148,51;
				end;
			}
			mes "[机场 工作人员]";
			mes "对不起, 但是您";
			mes "并没有飞空艇";
			mes "免费券, 同时您也";
			mes "没有足够的 zeny 来乘坐";
			mes "飞空艇.请记得";
			mes "乘坐价格为 1,200 zeny";
			close;
		}
	}
	mes "[机场 工作人员]";
	mes "谢谢!";
	mes "祝您愉快!";
	close;
}
