function	script	F_AuctionWarper	{
	mes "[拍卖行接待]";
	mes "您好!尊敬的冒险家";
	mes "您想进入";
	mes "拍卖大厅吗?";
	next;
	if (select("想去:不想") == 1) {
		if (getarg(0) == 1 || getarg(0) == 4) {
			mes "[拍卖行接待]";
			mes "太好了.那好.";
			mes "我希望您玩得开心";
			mes "祝您愉快~";
		} else {
			mes "[拍卖行接待]";
			mes "请上车.";
		}
		close2;
		switch(getarg(0)) {
		case 1:
			warp "auction_01",179,53; end;
		case 2:
			warp "auction_01",21,43; end;
		case 3:
			warp "auction_02",151,23; end;
		case 4:
			warp "auction_02",43,24; end;
		}
	}
	mes "[拍卖行接待]";
	mes "那么好吧,";
	mes "再见.";
	mes "如果您改变了主意";
	mes "请随时来找我";
	mes "谢谢您的支持~";
	close;
}
