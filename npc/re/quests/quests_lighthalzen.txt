function	script	lhz4enter	{
	if (getmapusers("que_lhz") == 0 && mobcount("que_lhz","#controllhz4::OnMyMobDead") < 1) {
		if (getarg(0) == 0) {
			mes "- 有了贾科莫.吉罗拉莫的提醒";
			mes "你忍住呼吸";
			mes "开始穿过被水淹没的";
			mes "电梯. -";
		} else {
			mes "[科研人员]";
			mes "你是谁?";
			mes "................";
			emotion e_omg;
			next;
			mes "男子逃跑, 你需要追上他.";
		}
		if (lhz_dun04 > 1) {
			changequest 5110,5109;
			set lhz_dun04,1;
		}
		close2;
		// Check twice because of 'next'.
		if (getmapusers("que_lhz") == 0 && mobcount("que_lhz","#controllhz4::OnMyMobDead") < 1) {
			donpcevent "#lhz4mes1::OnEnable";
			warp "que_lhz",244,52;
		}
		end;
	}
	mes "现在有其他人在里面, 轮到你等着了.";
	close;
}
